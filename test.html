#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int T;
ll x,y;
int main(){
    scanf("%d",&T);
    while(T--){
        scanf("%lld%lld",&x,&y);
        ll ans=0;
        ll sq=sqrt(x);
        if(y<=sq){
            // 枚举b从1到y求sum(b-1)=>等差数列求和
            ans=y*(y-1)/2;
        }else{
            // 等差数列求第一部分
            ans+=sq*(sq-1)/2;
            // 整除分块求第二部分
            for(ll l=sq+2,r;l<=y+1;l=r+1){
                if(x/l==0){
                    break;
                }
                r=min(y+1,x/(x/l));
                // printf("%d %d %d %d\n",l,r,r-l+1,x/l);
                ans+=(r-l+1)*(x/l);
            }
        }
        printf("%lld\n",ans);
    }
    return 0;
}
